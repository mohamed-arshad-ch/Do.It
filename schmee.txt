// This is your Prisma schema file,

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mysql"
  url      = env("MYSQL_DATABASE_URI")
}

model users {
  uid       String  @id @unique @db.VarChar(255)
  username  String  @unique
  createdAt String  @db.VarChar(255)
  active    Boolean @default(true)


  ledgers      ledgers[]
  transactions transactions[]
}

model ledgers {
  id          Int    @id @unique @default(autoincrement())
  user        users  @relation(fields: [uid], references: [uid])
  uid         String
  ledger_name String
  description String @db.VarChar(255)
  createdAt   String @db.VarChar(255)
  group_name  String @db.VarChar(50)


}

model transactions {
  t_id           Int    @id @unique @default(autoincrement())
  user           users  @relation(fields: [uid], references: [uid])
  uid            String
  group_name     String @db.VarChar(50)
  t_type         String @db.VarChar(20)
  t_catagory     String @db.VarChar(20)
  t_note         String @db.VarChar(1000)
  t_date         String @db.VarChar(10)
  t_amount       Float
  t_created_date String


}
